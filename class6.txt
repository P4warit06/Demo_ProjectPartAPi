  );
});

});

});
const count = ref(1);

  <div>
    <!-- ลบ components ถ้า onMounted -->
    <Header v-if="count <=3"/>

    <div class="">
      <p>Message: {{ msg }}</p>
      <button
        class="bg-purple-600 text-white rounded-md px-3 py-1"
        @click="reverseMsg"
      >
        Reverse
      </button>
    </div>
    <div class="flex space-x-1">
      <button
        @click="count--"
        class="border border-red-600 rounded-md px-2 py-1"
      >
        -
      </button>
      <p>Counter: {{ count }}</p>
      <button
        @click="count++"
